# gh-ost Task Run Log Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a dedicated gh-ost migration task-run log section with start/end timing, statement context, and error reporting.

**Architecture:** Extend the store log proto with `GHOST_MIGRATION_START` / `GHOST_MIGRATION_END`, expose them through the v1 API as one collapsed `GHOST_MIGRATION` timed entry, and emit the pair around `migrator.Migrate()`. The React task-run log viewer should treat gh-ost like other timed sections while using a distinct localized label.

**Tech Stack:** Go, protobuf/buf, Connect/gRPC generated code, React, TypeScript, i18next, Vitest.

func (o *ExecuteOptions) LogGhostMigrationStart(statement string) {
	if o == nil || o.CreateTaskRunLog == nil {
		return
	}
	err := o.CreateTaskRunLog(time.Now(), &storepb.TaskRunLog{
		Type: storepb.TaskRunLog_GHOST_MIGRATION_START,
		GhostMigrationStart: &storepb.TaskRunLog_GhostMigrationStart{
			Statement: statement,
		},
	})
	if err != nil {
		slog.Warn("failed to log gh-ost migration start", log.BBError(err))
	}
}

func (o *ExecuteOptions) LogGhostMigrationEnd(rerr string) {
	if o == nil || o.CreateTaskRunLog == nil {
		return
	}
	err := o.CreateTaskRunLog(time.Now(), &storepb.TaskRunLog{
		Type: storepb.TaskRunLog_GHOST_MIGRATION_END,
		GhostMigrationEnd: &storepb.TaskRunLog_GhostMigrationEnd{
			Error: rerr,
		},
	})
	if err != nil {
		slog.Warn("failed to log gh-ost migration end", log.BBError(err))
	}
}
