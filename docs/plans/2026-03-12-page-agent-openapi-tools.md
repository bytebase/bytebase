# Page Agent: OpenAPI-Based search_api & call_api

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace the static 60-operation hardcoded list with a build-time generated OpenAPI index, porting the MCP backend's multi-mode search and validated call patterns to the frontend.

**Architecture:** A Node.js codegen script reads `backend/api/mcp/gen/openapi.yaml` at build time and emits a compact TypeScript module with endpoint info + schema data. The frontend `search_api` and `call_api` tools consume this generated module, matching the MCP backend's search modes (list/browse/query/detail/schema) and validated execution.

**Tech Stack:** Node.js + js-yaml (codegen), TypeScript (runtime), Vite (build integration)

**Step 1: Create the codegen script**

```js
// frontend/scripts/generate_openapi_index.js
//
// Reads backend/api/mcp/gen/openapi.yaml and generates a TypeScript module
// with endpoint info + schema data for the page agent tools.
//
// Usage: node frontend/scripts/generate_openapi_index.js

const fs = require("fs");
const path = require("path");
const yaml = require("js-yaml");

const SPEC_PATH = path.resolve(
  __dirname,
  "../../backend/api/mcp/gen/openapi.yaml"
);
const OUT_PATH = path.resolve(
  __dirname,
  "../src/plugins/agent/logic/tools/gen/openapi-index.ts"
);

const spec = yaml.load(fs.readFileSync(SPEC_PATH, "utf8"));

// --- Extract endpoints ---

const endpoints = [];

for (const [pathStr, pathItem] of Object.entries(spec.paths || {})) {
  const op = pathItem.post;
  if (!op) continue;

  // /bytebase.v1.SQLService/Query -> service=SQLService, method=Query
  const trimmed = pathStr.replace(/^\/bytebase\.v1\./, "");
  const parts = trimmed.split("/");
  if (parts.length !== 2) continue;

  const [service, method] = parts;

  // Extract request/response schema refs
  let requestSchemaRef = "";
  let responseSchemaRef = "";

  const reqContent = op.requestBody?.content?.["application/json"];
  if (reqContent?.schema?.$ref) {
    requestSchemaRef = reqContent.schema.$ref;
  }

  const respContent = op.responses?.["200"]?.content?.["application/json"];
  if (respContent?.schema?.$ref) {
    responseSchemaRef = respContent.schema.$ref;
  }

  endpoints.push({
    operationId: op.operationId,
    path: pathStr,
    service,
    method,
    summary: op.summary || "",
    requestSchemaRef,
    responseSchemaRef,
  });
}

// --- Extract schemas (properties only, compact) ---

const schemas = {};
const componentSchemas = spec.components?.schemas || {};

for (const [name, schema] of Object.entries(componentSchemas)) {
  // Skip non-bytebase schemas (connect.error, etc.)
  if (!name.startsWith("bytebase.v1.")) continue;

  // Enum types
  if (schema.enum) {
    schemas[name] = {
      type: "enum",
      values: schema.enum,
    };
    continue;
  }

  // Object types with properties
  if (schema.properties) {
    const required = new Set(schema.required || []);
    const props = [];

    for (const [propName, prop] of Object.entries(schema.properties)) {
      let propType = prop.type || "object";

      // Handle $ref
      if (prop.$ref) {
        propType = prop.$ref.split("/").pop();
      }

      // Handle array items
      if (propType === "array" && prop.items) {
        if (prop.items.$ref) {
          propType = `array<${prop.items.$ref.split("/").pop()}>`;
        } else if (prop.items.type) {
          propType = `array<${prop.items.type}>`;
        }
      }

      props.push({
        name: propName,
        type: propType,
        ...(prop.description ? { description: prop.description } : {}),
        ...(required.has(propName) ? { required: true } : {}),
      });
    }

    if (props.length > 0) {
      schemas[name] = { type: "object", properties: props };
    }
  }
}

// --- Generate TypeScript ---

const output = `// AUTO-GENERATED by scripts/generate_openapi_index.js — DO NOT EDIT
// Source: backend/api/mcp/gen/openapi.yaml

export interface EndpointInfo {
  operationId: string;
  path: string;
  service: string;
  method: string;
  summary: string;
  requestSchemaRef: string;
  responseSchemaRef: string;
}

export interface PropertyInfo {
  name: string;
  type: string;
  description?: string;
  required?: boolean;
}

export interface SchemaInfo {
  type: "object" | "enum";
  properties?: PropertyInfo[];
  values?: string[];
}

export const endpoints: EndpointInfo[] = ${JSON.stringify(endpoints, null, 2)};

export const schemas: Record<string, SchemaInfo> = ${JSON.stringify(schemas, null, 2)};
`;

// Ensure output directory exists
fs.mkdirSync(path.dirname(OUT_PATH), { recursive: true });
fs.writeFileSync(OUT_PATH, output, "utf8");

console.log(
  `Generated ${OUT_PATH}: ${endpoints.length} endpoints, ${Object.keys(schemas).length} schemas`
);
```

**Step 1: Rewrite searchApi.ts with OpenAPIIndex class**

Port the MCP backend's search logic:

- Multi-mode search matching `tool_search.go`: list services, browse service, keyword query with scoring, endpoint detail, schema lookup
- Same scoring algorithm: keyword match + method/service/operationId/summary substring boosts + CRUD boost

```ts
// frontend/src/plugins/agent/logic/tools/searchApi.ts

import {
  endpoints as rawEndpoints,
  schemas,
  type EndpointInfo,
  type SchemaInfo,
  type PropertyInfo,
} from "./gen/openapi-index";

// --- Keyword extraction (port of openapi_index.go) ---

const STOP_WORDS = new Set([
  "the", "a", "an", "and", "or", "is", "are", "was", "were",
  "in", "on", "at", "to", "for", "of", "with", "by", "from",
  "service", "request", "response",
]);

function splitCamelCase(s: string): string[] {
  const words: string[] = [];
  let current = "";
  for (const ch of s) {
    if (ch >= "A" && ch <= "Z") {
      if (current) words.push(current);
      current = ch;
    } else if ((ch >= "a" && ch <= "z") || (ch >= "0" && ch <= "9")) {
      current += ch;
    } else {
      if (current) words.push(current);
      current = "";
    }
  }
  if (current) words.push(current);
  return words;
}

function extractKeywords(...texts: string[]): string[] {
  const set = new Set<string>();
  for (const text of texts) {
    for (const word of splitCamelCase(text)) {
      const lower = word.toLowerCase();
      if (lower.length >= 2 && !STOP_WORDS.has(lower)) set.add(lower);
    }
    for (const word of text.split(/[^a-zA-Z0-9]+/)) {
      const lower = word.toLowerCase();
      if (lower.length >= 2 && !STOP_WORDS.has(lower)) set.add(lower);
    }
  }
  return [...set];
}

const PRIMARY_CRUD = new Set([
  "List", "Get", "Create", "Update", "Delete", "Search", "Query", "Execute",
]);

function isPrimaryCRUD(method: string): boolean {
  for (const op of PRIMARY_CRUD) {
    if (method.startsWith(op)) return true;
  }
  return false;
}

// --- Type descriptions (port of openapi_index.go) ---

const TYPE_DESCRIPTIONS: Record<string, string> = {
  "google.protobuf.Timestamp": 'ISO 8601 format, e.g. "2024-01-15T01:30:15Z"',
  "google.protobuf.Duration": 'e.g. "3.5s" or "1h30m"',
  "google.protobuf.FieldMask": 'e.g. "title,engine"',
  "google.protobuf.Empty": "empty message",
  "google.protobuf.Any": "any JSON value",
  "google.protobuf.Struct": "JSON object",
  "google.protobuf.Value": "any JSON value",
};

// --- OpenAPIIndex class ---

class OpenAPIIndex {
  private byOperation = new Map<string, EndpointInfo>();
  private byService = new Map<string, EndpointInfo[]>();
  private keywords = new Map<string, EndpointInfo[]>();
  private services: string[] = [];

  constructor(private endpoints: EndpointInfo[]) {
    const serviceSet = new Set<string>();

    for (const ep of endpoints) {
      this.byOperation.set(ep.operationId, ep);

      const list = this.byService.get(ep.service) ?? [];
      list.push(ep);
      this.byService.set(ep.service, list);
      serviceSet.add(ep.service);

      for (const kw of extractKeywords(ep.service, ep.method, ep.summary)) {
        const kwList = this.keywords.get(kw) ?? [];
        kwList.push(ep);
        this.keywords.set(kw, kwList);
      }
    }

    this.services = [...serviceSet].sort();
  }

  getServices(): string[] {
    return this.services;
  }

  getEndpoint(operationId: string): EndpointInfo | undefined {
    // Try full format first
    const ep = this.byOperation.get(operationId);
    if (ep) return ep;

    // Try short format: Service/Method -> bytebase.v1.Service.Method
    const parts = operationId.split("/");
    if (parts.length === 2) {
      return this.byOperation.get(`bytebase.v1.${parts[0]}.${parts[1]}`);
    }
    return undefined;
  }

  getServiceEndpoints(service: string): EndpointInfo[] {
    return this.byService.get(service) ?? [];
  }

  search(query: string, limit = 5): EndpointInfo[] {
    const queryKeywords = extractKeywords(query);
    if (queryKeywords.length === 0) return [];

    const scores = new Map<EndpointInfo, number>();

    // Keyword matching
    for (const kw of queryKeywords) {
      for (const ep of this.keywords.get(kw) ?? []) {
        scores.set(ep, (scores.get(ep) ?? 0) + 1);
      }
    }

    // Substring matching with boosts
    const queryLower = query.toLowerCase();
    for (const ep of this.endpoints) {
      const methodLower = ep.method.toLowerCase();
      const serviceLower = ep.service.toLowerCase();
      let matched = false;

      if (methodLower === queryLower) {
        scores.set(ep, (scores.get(ep) ?? 0) + 10);
        matched = true;
      } else if (methodLower.includes(queryLower)) {
        scores.set(ep, (scores.get(ep) ?? 0) + 5);
        matched = true;
      }

      if (serviceLower.includes(queryLower)) {
        scores.set(ep, (scores.get(ep) ?? 0) + 4);
        matched = true;
      }

      if (ep.operationId.toLowerCase().includes(queryLower)) {
        scores.set(ep, (scores.get(ep) ?? 0) + 3);
        matched = true;
      }

      if (ep.summary.toLowerCase().includes(queryLower)) {
        scores.set(ep, (scores.get(ep) ?? 0) + 2);
        matched = true;
      }

      if (matched && isPrimaryCRUD(ep.method)) {
        scores.set(ep, (scores.get(ep) ?? 0) + 2);
      }
    }

    // Sort by score descending
    const results = [...scores.entries()]
      .sort((a, b) => b[1] - a[1])
      .map(([ep]) => ep);

    return limit > 0 ? results.slice(0, limit) : results;
  }

  searchInService(query: string, service: string, limit = 5): EndpointInfo[] {
    return this.search(query, 0)
      .filter((ep) => ep.service === service)
      .slice(0, limit > 0 ? limit : Infinity);
  }
}

// --- Singleton index ---

const index = new OpenAPIIndex(rawEndpoints);

// --- Formatting (port of tool_search.go) ---

function formatServiceList(): string {
  const services = index.getServices();
  let text = "## Available Services\n\n";
  text += 'Use `search_api(service="ServiceName")` to list endpoints in a service.\n\n';

  for (const svc of services) {
    const eps = index.getServiceEndpoints(svc);
    text += `- **${svc}** (${eps.length} endpoints)\n`;
  }

  text += `\nTotal: ${services.length} services\n`;
  return text;
}

function formatEndpoints(eps: EndpointInfo[], limit: number): string {
  let text = "";
  if (limit > 0 && eps.length > limit) {
    text += `Showing ${limit} of ${eps.length} results:\n\n`;
    eps = eps.slice(0, limit);
  } else {
    text += `Found ${eps.length} endpoints:\n\n`;
  }

  eps.forEach((ep, i) => {
    text += `### ${i + 1}. ${ep.service}/${ep.method}\n`;
    text += `${ep.summary}\n\n`;
  });

  return text;
}

function formatProperty(prop: PropertyInfo): string {
  const req = prop.required ? " (required)" : "";
  let desc = "";
  if (TYPE_DESCRIPTIONS[prop.type]) {
    desc = ` // ${TYPE_DESCRIPTIONS[prop.type]}`;
  } else if (prop.description) {
    const clean = prop.description.replace(/\n/g, " ").slice(0, 100);
    desc = ` // ${clean}${prop.description.length > 100 ? "..." : ""}`;
  }
  return `  "${prop.name}": ${prop.type}${req}${desc}\n`;
}

function formatEndpointDetail(operationId: string): string {
  const ep = index.getEndpoint(operationId);
  if (!ep) {
    return `Unknown operationId: ${operationId}\n\nUse search_api(query="...") to find valid operations.`;
  }

  let text = `## ${ep.service}/${ep.method}\n\n${ep.summary}\n\n`;

  // Request schema
  if (ep.requestSchemaRef) {
    const schemaName = ep.requestSchemaRef.split("/").pop()!;
    const schema = schemas[schemaName];
    if (schema?.properties) {
      text += "### Request Body\n```json\n{\n";
      for (const prop of schema.properties) text += formatProperty(prop);
      text += "}\n```\n\n";
    }
  }

  // Response schema
  if (ep.responseSchemaRef) {
    const schemaName = ep.responseSchemaRef.split("/").pop()!;
    const schema = schemas[schemaName];
    if (schema?.properties) {
      text += "### Response Body\n```json\n{\n";
      for (const prop of schema.properties) text += formatProperty(prop);
      text += "}\n```\n";
    }
  }

  return text;
}

function formatSchemaDetail(schemaName: string): string {
  const fullName = schemaName.startsWith("bytebase.v1.")
    ? schemaName
    : `bytebase.v1.${schemaName}`;

  const schema = schemas[fullName] ?? schemas[schemaName];
  if (!schema) {
    return `Unknown schema: ${schemaName}\n\nUse search_api(operationId="...") to see schemas in request/response bodies.`;
  }

  let text = `## ${fullName}\n\n`;

  if (schema.type === "enum" && schema.values) {
    text += `**Enum values:** ${schema.values.join(", ")}\n`;
    return text;
  }

  if (schema.properties) {
    for (const prop of schema.properties) text += formatProperty(prop);
  }

  return text;
}

// --- Public API (matches MCP tool_search.go modes) ---

export interface SearchApiArgs {
  operationId?: string;
  schema?: string;
  query?: string;
  service?: string;
  limit?: number;
}

export function getEndpointPath(operationId: string): string | undefined {
  return index.getEndpoint(operationId)?.path;
}

export async function searchApi(args: SearchApiArgs): Promise<string> {
  const limit = Math.min(Math.max(args.limit ?? 5, 1), 50);

  // Detail mode
  if (args.operationId) {
    return formatEndpointDetail(args.operationId);
  }

  // Schema lookup
  if (args.schema) {
    return formatSchemaDetail(args.schema);
  }

  // List all services
  if (!args.query && !args.service) {
    return formatServiceList();
  }

  // Browse service (no query)
  if (args.service && !args.query) {
    const eps = index.getServiceEndpoints(args.service);
    if (eps.length === 0) {
      return `No endpoints found for service: ${args.service}\n\nAvailable services:\n${formatServiceList()}`;
    }
    return formatEndpoints(eps, 0);
  }

  // Search within service
  if (args.service && args.query) {
    const eps = index.searchInService(args.query, args.service, limit);
    if (eps.length === 0) {
      return `No endpoints found for query "${args.query}" in service ${args.service}\n\nTry:\n- Different keywords\n- Browsing the service with search_api(service="${args.service}")`;
    }
    return formatEndpoints(eps, limit);
  }

  // Query search
  const eps = index.search(args.query!, limit);
  if (eps.length === 0) {
    return `No endpoints found for query: "${args.query}"\n\nTry:\n- Different keywords\n- Listing services with search_api() (no parameters)\n- Browsing a service with search_api(service="ServiceName")`;
  }
  return formatEndpoints(eps, limit);
}

// frontend/src/plugins/agent/logic/tools/callApi.ts

import { getEndpointPath } from "./searchApi";

export interface CallApiArgs {
  operationId: string;
  body?: Record<string, unknown>;
}

export async function callApi(args: CallApiArgs): Promise<string> {
  if (!args.operationId) {
    return JSON.stringify({
      error: "operationId is required",
    });
  }

  const path = getEndpointPath(args.operationId);
  if (!path) {
    return JSON.stringify({
      error: `Unknown operation: ${args.operationId}. Use search_api to find valid operations.`,
    });
  }

  try {
    const response = await fetch(`${window.location.origin}${path}`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Connect-Protocol-Version": "1",
      },
      credentials: "include",
      body: JSON.stringify(args.body ?? {}),
    });

    const data = await response.json().catch(() => null);

    if (!response.ok) {
      let errorMsg = `HTTP ${response.status}`;
      if (data && typeof data === "object") {
        if ("message" in data) errorMsg = data.message as string;
        else if ("code" in data) errorMsg = data.code as string;
      }
      return JSON.stringify({
        status: response.status,
        error: errorMsg,
        response: data,
      });
    }

    return JSON.stringify({
      status: response.status,
      response: data,
    });
  } catch (err) {
    return JSON.stringify({
      error: `Failed to call ${args.operationId}: ${err instanceof Error ? err.message : String(err)}`,
    });
  }
}
